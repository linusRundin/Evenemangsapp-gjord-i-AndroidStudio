79def decode_token(encoded_token, csrf_value=None, allow_expired=False): 

80 """ 

81 Returns the decoded token (python dict) from an encoded JWT. This does all 

82 the checks to insure that the decoded token is valid before returning it. 

83 

84 :param encoded_token: The encoded JWT to decode into a python dict. 

85 :param csrf_value: Expected CSRF double submit value (optional) 

86 :param allow_expired: Options to ignore exp claim validation in token 

87 :return: Dictionary containing contents of the JWT 

88 """ 

89 jwt_manager = _get_jwt_manager() 

90 unverified_claims = jwt.decode( 

91 encoded_token, verify=False, algorithms=config.decode_algorithms 

92 ) 

93 unverified_headers = jwt.get_unverified_header(encoded_token) 

94 # Attempt to call callback with both claims and headers, but fallback to just claims 

95 # for backwards compatibility 

96 try: 

97 secret = jwt_manager._decode_key_callback(unverified_claims, unverified_headers) 

98 except TypeError: 

99 msg = ( 

100 "The single-argument (unverified_claims) form of decode_key_callback ", 

101 "is deprecated. Update your code to use the two-argument form ", 

102 "(unverified_claims, unverified_headers)." 

103 ) 

104 warn(msg, DeprecationWarning) 

105 secret = jwt_manager._decode_key_callback(unverified_claims) 

106 

107 try: 

108 return decode_jwt( 

109 encoded_token=encoded_token, 

110 secret=secret, 

111 algorithms=config.decode_algorithms, 

112 identity_claim_key=config.identity_claim_key, 

113 user_claims_key=config.user_claims_key, 

114 csrf_value=csrf_value, 

115 audience=config.audience, 

116 issuer=config.issuer, 

117 leeway=config.leeway, 

118 allow_expired=allow_expired 

119 ) 

120 except ExpiredSignatureError: 

121 expired_token = decode_jwt( 

122 encoded_token=encoded_token, 

123 secret=secret, 

124 algorithms=config.decode_algorithms, 

125 identity_claim_key=config.identity_claim_key, 

126 user_claims_key=config.user_claims_key, 

127 csrf_value=csrf_value, 

128 audience=config.audience, 

129 issuer=config.issuer, 

130 leeway=config.leeway, 

131 allow_expired=True 

132 ) 

133 ctx_stack.top.expired_jwt = expired_token 

134 raise

145def create_access_token(identity, fresh=False, expires_delta=None, user_claims=None, 

146 headers=None): 

147 """ 

148 Create a new access token. 

149 

150 :param identity: The identity of this token, which can be any data that is 

151 json serializable. It can also be a python object, in which 

152 case you can use the 

153 :meth:`~flask_jwt_extended.JWTManager.user_identity_loader` 

154 to define a callback function that will be used to pull a 

155 json serializable identity out of the object. 

156 :param fresh: If this token should be marked as fresh, and can thus access 

157 :func:`~flask_jwt_extended.fresh_jwt_required` endpoints.

160 this token will be considered fresh. 

161 :param expires_delta: A `datetime.timedelta` for how long this token should

163 expiration. If this is None, it will use the 

164 'JWT_ACCESS_TOKEN_EXPIRES` config value 

165 (see :ref:`Configuration Options`) 

166 :param user_claims: Optional JSON serializable to override user claims. 

167 :param headers: Optional, valid dict for specifying additional headers in JWT 

168 header section 

169 :return: An encoded access token 

170 """ 

171 jwt_manager = _get_jwt_manager() 

172 return jwt_manager._create_access_token(identity, fresh, expires_delta, user_claims, 

173 headers=headers)

260def set_access_cookies(response, encoded_access_token, max_age=None): 

261 """ 

262 Takes a flask response object, and an encoded access token, and configures 

263 the response to set in the access token in a cookie. If `JWT_CSRF_IN_COOKIES` 

264 is `True` (see :ref:`Configuration Options`), this will also set the CSRF 

265 double submit values in a separate cookie. 

266 

267 :param response: The Flask response object to set the access cookies in. 

268 :param encoded_access_token: The encoded access token to set in the cookies. 

269 :param max_age: The max age of the cookie. If this is None, it will use the 

270 `JWT_SESSION_COOKIE` option (see :ref:`Configuration Options`). 

271 Otherwise, it will use this as the cookies `max-age` and the 

272 JWT_SESSION_COOKIE option will be ignored. Values should be 

273 the number of seconds (as an integer). 

274 """ 

275 if not config.jwt_in_cookies: 

276 raise RuntimeWarning("set_access_cookies() called without " 

277 "'JWT_TOKEN_LOCATION' configured to use cookies") 

278 

279 # Set the access JWT in the cookie 

280 response.set_cookie(config.access_cookie_name, 

281 value=encoded_access_token, 

282 max_age=max_age or config.cookie_max_age, 

283 secure=config.cookie_secure, 

284 httponly=True, 

285 domain=config.cookie_domain, 

286 path=config.access_cookie_path, 

287 samesite=config.cookie_samesite) 

288 

289 # If enabled, set the csrf double submit access cookie 

290 if config.csrf_protect and config.csrf_in_cookies: 

291 response.set_cookie(config.access_csrf_cookie_name, 

292 value=get_csrf_token(encoded_access_token), 

293 max_age=max_age or config.cookie_max_age, 

294 secure=config.cookie_secure, 

295 httponly=False, 

296 domain=config.cookie_domain, 

297 path=config.access_csrf_cookie_path, 

298 samesite=config.cookie_samesite)
