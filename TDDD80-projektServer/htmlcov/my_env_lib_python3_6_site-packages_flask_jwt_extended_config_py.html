13# Older versions of pyjwt do not have the requires_cryptography set. Also, 

14# older versions will not be adding new algorithms to them, so I can hard code 

15# the default version here and be safe. If there is a newer algorithm someone 

16# wants to use, they will need newer versions of pyjwt and it will be included 

17# in their requires_cryptography set, and if they attempt to use it in older 

18# versions of pyjwt, it will kick it out as an unrecognized algorithm. 

19try: 

20 from jwt.algorithms import requires_cryptography 

21except ImportError: # pragma: no cover 

22 requires_cryptography = {'RS256', 'RS384', 'RS512', 'ES256', 'ES384', 

23 'ES521', 'ES512', 'PS256', 'PS384', 'PS512'}
